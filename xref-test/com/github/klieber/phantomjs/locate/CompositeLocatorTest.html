21  package com.github.klieber.phantomjs.locate;
22  
23  import org.junit.Before;
24  import org.junit.Test;
25  import org.junit.runner.RunWith;
26  import org.mockito.Mock;
27  import org.mockito.runners.MockitoJUnitRunner;
28  
29  import java.util.Arrays;
30  
31  import static org.junit.Assert.assertEquals;
32  import static org.junit.Assert.assertNull;
33  import static org.mockito.Mockito.*;
34  
35  @RunWith(MockitoJUnitRunner.class)
36  public class CompositeLocatorTest {
37  
38    private static final String LOCATION_A = "/opt/phantomjs/bin/phantomjs";
39    private static final String LOCATION_B = "/usr/bin/phantomjs";
40  
41    @Mock
42    private Locator locatorA;
43  
44    @Mock
45    private Locator locatorB;
46  
47    private CompositeLocator compositeLocator;
48  
49    @Before
50    public void before() {
51      compositeLocator = new CompositeLocator(Arrays.asList(locatorA,locatorB));
52    }
53  
54    @Test
55    public void shouldLocateLocationA() throws Exception  {
56      when(locatorA.locate()).thenReturn(LOCATION_A);
57      assertEquals(LOCATION_A, compositeLocator.locate());
58      verifyNoMoreInteractions(locatorB);
59    }
60  
61    @Test
62    public void shouldLocateLocationB() throws Exception  {
63      when(locatorB.locate()).thenReturn(LOCATION_B);
64      assertEquals(LOCATION_B, compositeLocator.locate());
65      verify(locatorA).locate();
66    }
67  
68    @Test
69    public void shouldNotLocateAnything() throws Exception  {
70      assertNull(compositeLocator.locate());
71      verify(locatorA).locate();
72      verify(locatorB).locate();
73    }
74  }
